Änderung: Sprach-Check per Klick statt „erledigt“

In deiner index.html den kompletten JavaScript-Bereich von <script> bis </script> durch folgenden Block ersetzen:

<script>
const checkboxes = document.querySelectorAll("input[type=checkbox]");
const progressBar = document.getElementById("progressBar");
const progressText = document.getElementById("progressText");
const voiceStatus = document.getElementById("voiceStatus");

let guideItems = [];
let currentGuideIndex = -1;
let isVoiceGuideRunning = false;

function updateProgress() {
  let checked = 0;

  checkboxes.forEach((cb, index) => {
    localStorage.setItem("wohnmobil_check_" + index, cb.checked);
    cb.closest(".item").classList.toggle("done", cb.checked);
    if (cb.checked) checked++;
  });

  const percent = Math.round((checked / checkboxes.length) * 100);
  progressBar.style.width = percent + "%";
  progressText.textContent = percent + "%";
}

checkboxes.forEach((cb, index) => {
  cb.checked = localStorage.getItem("wohnmobil_check_" + index) === "true";

  cb.addEventListener("change", () => {
    updateProgress();

    if (isVoiceGuideRunning && currentGuideIndex >= 0) {
      const currentItem = guideItems[currentGuideIndex];

      if (currentItem && cb === currentItem.checkbox && cb.checked) {
        currentItem.label.classList.remove("active");
        setTimeout(nextGuideItem, 350);
      }
    }
  });
});

function cleanForSpeech(text) {
  return text
    .replace(/[🚐🐶🚦🏠🔌🏕️✅🎤🌳]/g, "")
    .replace(/Kapitel:\s*/g, "")
    .replace(/\s+/g, " ")
    .trim();
}

function speak(text, afterSpeak) {
  if (!("speechSynthesis" in window)) {
    alert("Dein Browser unterstützt leider keine Sprachausgabe.");
    return;
  }

  window.speechSynthesis.cancel();

  const speech = new SpeechSynthesisUtterance(cleanForSpeech(text));
  speech.lang = "de-DE";
  speech.rate = 0.85;
  speech.pitch = 1;
  speech.onend = function() {
    if (afterSpeak) afterSpeak();
  };

  window.speechSynthesis.speak(speech);
}

function buildGuideItems() {
  guideItems = [];

  document.querySelectorAll("main section").forEach(section => {
    const title = section.querySelector("h2") ? section.querySelector("h2").innerText.trim() : "";
    const items = section.querySelectorAll(".item");

    items.forEach(label => {
      const cb = label.querySelector("input[type=checkbox]");
      if (cb && !cb.checked) {
        guideItems.push({
          section: title,
          label: label,
          checkbox: cb,
          text: label.innerText.trim()
        });
      }
    });
  });
}

function clearActiveItems() {
  document.querySelectorAll(".item").forEach(item => item.classList.remove("active"));
}

function startVoiceGuide() {
  buildGuideItems();
  clearActiveItems();

  if (guideItems.length === 0) {
    voiceStatus.innerHTML = "✅ Alles erledigt. Gute Fahrt!";
    speak("Super. Alle Punkte sind erledigt. Reinhold, Uschi und Molly wünschen gute Fahrt!");
    return;
  }

  currentGuideIndex = -1;
  isVoiceGuideRunning = true;
  voiceStatus.innerHTML = "🎤 Sprach-Check läuft. Hake jeden vorgelesenen Punkt einfach per Klick auf das Kästchen ab.";
  speak("Sprach-Check gestartet. Hake jeden vorgelesenen Punkt einfach per Klick auf das Kästchen ab.", nextGuideItem);
}

function nextGuideItem() {
  clearActiveItems();
  currentGuideIndex++;

  if (currentGuideIndex >= guideItems.length) {
    isVoiceGuideRunning = false;
    voiceStatus.innerHTML = "✅ Sprach-Check beendet. Alle abgefragten Punkte sind erledigt.";
    speak("Sprach-Check beendet. Alle abgefragten Punkte sind erledigt. Gute Fahrt!");
    updateProgress();
    return;
  }

  const item = guideItems[currentGuideIndex];

  if (item.checkbox.checked) {
    nextGuideItem();
    return;
  }

  item.label.classList.add("active");
  item.label.scrollIntoView({ behavior: "smooth", block: "center" });

  const spoken = "Jetzt: " + item.section + ". " + item.text + ".";
  voiceStatus.innerHTML = "Aktueller Punkt:<br><b>" + item.section + "</b><br>" + item.text + "<br><br>Bitte Kästchen anklicken, wenn erledigt.";
  speak(spoken);
}

function speakChecklist() {
  buildGuideItems();

  if (guideItems.length === 0) {
    speak("Super. Alle Punkte der Wohnmobil Checkliste sind erledigt. Reinhold Uschi und Molly wünschen gute Fahrt!");
  } else {
    let grouped = {};
    guideItems.forEach(item => {
      if (!grouped[item.section]) grouped[item.section] = [];
      grouped[item.section].push(item.text);
    });

    let text = "Es sind noch " + guideItems.length + " Punkte offen. ";
    Object.keys(grouped).forEach(section => {
      text += section + ". ";
      text += grouped[section].join(". ") + ". ";
    });
    speak(text);
  }
}

function resetChecklist() {
  if (confirm("Wirklich alle Häkchen zurücksetzen?")) {
    window.speechSynthesis.cancel();
    isVoiceGuideRunning = false;
    currentGuideIndex = -1;

    checkboxes.forEach((cb, index) => {
      cb.checked = false;
      localStorage.removeItem("wohnmobil_check_" + index);
    });

    clearActiveItems();
    updateProgress();
    voiceStatus.innerHTML = "Drücke <b>„Sprach-Check starten“</b>. Die App liest jeden offenen Punkt vor. Den Punkt erledigst du einfach per Klick auf das Kästchen.";
  }
}

updateProgress();
</script>
